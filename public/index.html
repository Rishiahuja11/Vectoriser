const btn = document.getElementById('convertBtn'); const loader = document.getElementById('loader'); const input = document.getElementById('fileInput'); const preview = document.getElementById('preview'); input.onchange = (e) => { const reader = new FileReader(); reader.onload = (event) => { preview.src = event.target.result; preview.classList.remove('hidden'); document.getElementById('uploadText').classList.add('hidden'); }; reader.readAsDataURL(e.target.files[0]); }; btn.onclick = () => { if (!preview.src) return alert("Please upload an image first!"); loader.classList.remove('hidden'); // Use ImageTracer to convert the image to colorful SVG // 'posterized2' is a preset that keeps colors vibrant ImageTracer.imageToSVG( preview.src, function(svgstr){ // Create a download link for the SVG const blob = new Blob([svgstr], {type: 'image/svg+xml'}); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'vectorized-image.svg'; a.click(); loader.classList.add('hidden'); }, 'posterized2' ); };
